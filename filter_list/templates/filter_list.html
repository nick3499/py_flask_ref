filter_list.py

      
import json
import requests
from flask import Flask, render_template

app = Flask(__name__)

@app.route('/')
def filter_list():
    l = [1, 2, 'a', 'b']
    f = list(filter(lambda x: type(x) == int, l))
    return render_template("filter_list.html", filtered=f)

if __name__ == '__main__':
    app.run(debug=True)
